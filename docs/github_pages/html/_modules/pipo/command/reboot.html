from discord.ext.commands import Context as Dctx

from pipo.command.command import Command

@dataclass
class Reboot(Command):
    """Command to reboot bot."""

    bot: Pipo
    ctx: Dctx

    async def _execute(self) -> None:
        await self.bot.reboot(self.ctx)
